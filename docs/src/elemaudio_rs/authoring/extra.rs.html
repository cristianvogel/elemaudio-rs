93/// Prototype convolution helper with fixed-size spectral IR edits.
94///
95/// Smoothing (FrameBlur) uses FrameLib `Time_Smoothing` semantics:
96/// - magnitudes flow through a per-bin moving average modelled on
97///   `FrameLib_MovingAverage` (separate `alpha_up` / `alpha_down` derived
98///   from `blurFrames` for asymmetric attack/release),
99/// - phases flow through a per-bin unwrapped accumulator that smooths
100///   phase *deltas* across frames (modelled on `FrameLib_FrameDelta` plus
101///   integration). This preserves phase coherence under heavy blur.
102///
103/// Props:
104/// - `path`: shared resource id for the impulse response
105/// - `partitionSize`: optional power-of-two IR edit partition size, rounded up
106/// - `magnitudeGainDb`: optional global spectral magnitude gain in dB
107/// - `maxBlurFrames`: optional capacity of the FrameBlur smoother in frames.
108///   Larger values cost more memory but allow longer windowed smoothing.
109///   A change triggers a convolver rebuild. Defaults to `16`.
110/// - `blurFrames`: optional active smoothing window in frames
111///   (`1..maxBlurFrames`). When non-zero this overrides the continuous
112///   `blur` child and pins the smoothing window. Set to `0` to hand
113///   control back to the `blur` child signal.
114///
115/// Child order:
116/// - `tiltDbPerOct`: frame-latched spectral tilt signal in dB/octave
117/// - `blur`: frame-latched 0..1 smoothing amount (mapped internally to a
118///   frame count and a phase-delta smoothing alpha)
119/// - `x`: audio input
120pub fn convolve_spectral(
121    props: serde_json::Value,
122    tilt_db_per_oct: impl Into<ElemNode>,
123    blur: impl Into<ElemNode>,
124    x: impl Into<ElemNode>,
125) -> Node {
126    Node::new(
127        "extra.convolveSpectral",
128        props,
129        vec![
130            resolve(tilt_db_per_oct),
131            resolve(blur),
132            resolve(x),
133        ],
134    )
135}

154/// Recursive foldback shaper helper.
155///
156/// Computes a recursive soft-saturation using fold-back distortion. The shape
157/// depends on the threshold and gain parameters.
158///
159/// For best performance with the fast-path update system, supply a `key` prefix.
160/// This allows threshold and amplitude to be updated **without rebuilding the graph**.
161///
162/// Props:
163/// - `key`: optional prefix for stable node identity; enables direct updates via
164///   `mounted.node_with_key("{key}_thresh")` and `mounted.node_with_key("{key}_amp")`
165/// - `thresh`: fold threshold, must be positive
166/// - `amp`: output gain, defaults to `1 / thresh`
167///
168/// # Example: Keyed foldback with parameter updates
169///
170/// ```ignore
171/// use elemaudio_rs::{Graph, el, extra};
172/// use serde_json::json;
173///
174/// // Create a graph with a keyed foldback
175/// let graph = Graph::new().render(
176///     extra::foldback(
177///         json!({
178///             "key": "shaper",
179///             "thresh": 0.5,
180///             "amp": 2.0,
181///         }),
182///         el::cycle(el::const_(440.0)),
183///     )
184/// );
185///
186/// // Mount the graph and get handles for direct updates
187/// let mounted = graph.mount();
188/// let batch = mounted.into_batch();
189/// runtime.execute(&batch);
190///
191/// // Later, update threshold WITHOUT rebuilding the entire graph
192/// if let Some(thresh_node) = mounted.node_with_key("shaper_thresh") {
193///     let update = thresh_node.set_const_value(0.7);
194///     runtime.execute(&update);
195/// }
196///
197/// // Similarly for amplitude
198/// if let Some(amp_node) = mounted.node_with_key("shaper_amp") {
199///     let update = amp_node.set_const_value(1.5);
200///     runtime.execute(&update);
201/// }
202/// ```
203pub fn foldback(props: serde_json::Value, x: impl Into<ElemNode>) -> Node {
204    let mut props = props;
205
206    // Extract and validate threshold
207    let thresh = match props.get("thresh").and_then(|v| v.as_f64()) {
208        Some(t) if t.is_finite() && t > 0.0 => t,
209        _ => {
210            log::error!(
211                "foldback: props missing required positive 'thresh' value, returning passthrough"
212            );
213            return resolve(x);
214        }
215    };
216
217    // Extract amplitude, defaulting to 1 / thresh
218    let amp = props
219        .get("amp")
220        .and_then(|value| value.as_f64())
221        .filter(|value| value.is_finite())
222        .unwrap_or(1.0 / thresh);
223
224    // Extract optional key prefix for fast-path updates
225    let key_prefix = props
226        .get("key")
227        .and_then(|value| value.as_str())
228        .map(|k| k.to_string());
229
230    // Remove control props before constructing the graph
231    if let serde_json::Value::Object(map) = &mut props {
232        map.remove("thresh");
233        map.remove("amp");
234        map.remove("key");
235    }
236
237    let x = resolve(x);
238
239    // Create const nodes with keys if a prefix was supplied
240    let thresh_node = match &key_prefix {
241        Some(prefix) => el::const_with_key(&format!("{}_thresh", prefix), thresh),
242        None => el::const_(thresh),
243    };
244
245    let amp_node = match &key_prefix {
246        Some(prefix) => el::const_with_key(&format!("{}_amp", prefix), amp),
247        None => el::const_(amp),
248    };
249
250    // Build the foldback computation graph
251    let folded = el::sub([
252        el::abs(el::sub([
253            el::abs(el::r#mod(
254                el::sub([x.clone(), thresh_node.clone()]),
255                el::mul([el::const_(4.0), thresh_node.clone()]),
256            )),
257            el::mul([el::const_(2.0), thresh_node.clone()]),
258        ])),
259        thresh_node.clone(),
260    ]);
261    let should_fold = el::ge(el::abs(x.clone()), thresh_node);
262
263    Node::new(
264        "mul",
265        props,
266        vec![amp_node, el::select(should_fold, folded, x)],
267    )
268}

270/// VariSlope SVF — cascaded Butterworth SVF with Rossum-style continuous slope
271/// morphing.
272///
273/// # Overview
274///
275/// This node exposes a continuously variable filter order (slope) that morphs
276/// smoothly between 1 and 6 cascaded second-order Butterworth SVF stages
277/// (12–72 dB/oct) at audio rate, inspired by Dave Rossum's analog cascade
278/// designs.
279///
280/// Q is fixed internally at Butterworth (√2 ≈ 1.414, maximally flat magnitude)
281/// and is not exposed. The slope is the sole tonal control: one knob that
282/// determines how aggressively the filter rolls off.
283///
284/// All six internal stages run every sample so their integrator states remain
285/// warm regardless of the current slope value. The output is a linear crossfade
286/// between the two adjacent integer-order outputs that bracket the current slope,
287/// so the filter order morphs without clicks, discontinuities, or dropout.
288///
289/// Per-stage gain correction (matched magnitude at cutoff) prevents the BLT
290/// passband droop from compounding across stages.
291///
292/// # Contrast with `el.svf`
293///
294/// The vendor `el.svf` is a single-stage Simper SVF (12 dB/oct fixed order)
295/// with an exposed Q parameter. `vari_slope_svf` removes Q and adds continuous
296/// Butterworth slope morphing as its defining feature.
297///
298/// # Inputs
299///
300/// | Index | Signal      | Required | Default  | Notes                         |
301/// |-------|-------------|----------|----------|-------------------------------|
302/// | 0     | `cutoff_hz` | yes      | —        | Cutoff frequency in Hz        |
303/// | 1     | `audio`     | yes      | —        | Audio input signal            |
304/// | 2     | `slope`     | no       | `1.0`    | Continuous order \[1.0, 6.0\] |
305///
306/// # Properties
307///
308/// | Key          | Type   | Values                              |
309/// |--------------|--------|-------------------------------------|
310/// | `filterType` | string | `"lowpass"` / `"lp"` (default)      |
311/// |              |        | `"highpass"` / `"hp"`               |
312///
313/// # Example
314///
315/// ```ignore
316/// use elemaudio_rs::{el, extra};
317/// use serde_json::json;
318///
319/// // Static 24 dB/oct lowpass at 800 Hz (slope = Some).
320/// let node = extra::vari_slope_svf(
321///     json!({ "filterType": "lowpass" }),
322///     el::const_(json!({ "value": 800.0 })),  // cutoff
323///     source,                                   // audio
324///     Some(el::const_(json!({ "value": 2.0 }))),// slope = 24 dB/oct
325/// );
326///
327/// // Default slope (12 dB/oct) — pass None.
328/// let node = extra::vari_slope_svf(
329///     json!({ "filterType": "lowpass" }),
330///     el::const_(json!({ "value": 800.0 })),
331///     source,
332///     None::<Node>,
333/// );
334///
335/// // Slope swept from 1.0 → 6.0 by an LFO for a dynamic order morph.
336/// let slope_lfo = el::add(
337///     el::const_(json!({ "value": 3.5 })),
338///     el::mul(el::const_(json!({ "value": 2.5 })),
339///             el::cycle(el::const_(json!({ "value": 0.25 })))),
340/// );
341/// let node = extra::vari_slope_svf(
342///     json!({ "filterType": "lowpass" }),
343///     cutoff, source, Some(slope_lfo),
344/// );
345/// ```
346pub fn vari_slope_svf(
347    props: serde_json::Value,
348    cutoff: impl Into<ElemNode>,
349    audio: impl Into<ElemNode>,
350    slope: Option<impl Into<ElemNode>>,
351) -> Node {
352    let mut children = vec![resolve(cutoff), resolve(audio)];
353    if let Some(s) = slope {
354        children.push(resolve(s));
355    }
356    Node::new("variSlopeSvf", props, children)
357}

795/// Stride delay with a feedback insert loop (mono).
796///
797/// The `insert` closure receives the **feedback audio signal** — the
798/// actual delayed audio coming back through the tap — and returns a
799/// processed version (e.g., filtered, pitch-shifted). The `fb` argument
800/// is the **feedback amount** (a gain coefficient) applied to the insert
801/// return before summing back into the delay input.
802///
803/// Requires `"fbtap"` in props — a unique name for the tapIn/tapOut pair.
804///
805/// # Props
806///
807/// Same as [`stride_delay`] plus:
808///
809/// | Key       | Type   | Required | Notes                              |
810/// |-----------|--------|----------|------------------------------------|
811/// | `fbtap`   | string | yes      | Unique tap name for feedback loop  |
812///
813/// # Example
814///
815/// ```ignore
816/// use elemaudio_rs::{el, extra};
817/// use serde_json::json;
818///
819/// // Delay with a lowpass filter darkening each repeat
820/// let delayed = extra::stride_delay_with_insert(
821///     json!({ "maxDelayMs": 1500, "transitionMs": 60, "fbtap": "fb_loop" }),
822///     el::const_with_key("delay", 250.0),    // delay time
823///     el::const_with_key("fb_amt", 0.5),     // feedback amount
824///     |fb_audio| {
825///         // fb_audio is the delayed signal coming back through the loop.
826///         // Filter it so each repeat gets darker.
827///         el::lowpass(el::const_(2000.0), el::const_(0.707), fb_audio)
828///     },
829///     el::r#in(json!({"channel": 0}), None), // audio input
830/// );
831/// ```
832///
833/// # Signal flow
834///
835/// ```text
836///                    ┌─── insert(fb_audio) ◄── tapIn(fbtap)
837///                    ▼
838/// fb_amount * insert_return ──┐
839///                              ▼
840/// audio_input ────────── add ──► stridedelay(internal fb=0)
841///                                       │
842///                               tapOut(fbtap) ──► output
843/// ```
844///
845/// Note: the feedback path has 1-block latency (tapIn/tapOut).
846pub fn stride_delay_with_insert(
847    props: serde_json::Value,
848    delay_ms: impl Into<ElemNode>,
849    fb: impl Into<ElemNode>,
850    insert: impl FnOnce(Node) -> Node,
851    x: impl Into<ElemNode>,
852) -> Node {
853    let mut props = props;
854
855    // Extract fbtap name from props. Fall back to stride_delay without insert.
856    let Some(fbtap) = props
857        .get("fbtap")
858        .and_then(|v| v.as_str())
859        .map(|s| s.to_string())
860    else {
861        log::error!(
862            "stride_delay_with_insert: missing 'fbtap' prop, falling back to stride_delay without insert"
863        );
864        return stride_delay(props, delay_ms, fb, x);
865    };
866
867    // Remove fbtap before passing to the native node.
868    if let serde_json::Value::Object(ref mut map) = props {
869        map.remove("fbtap");
870    }
871
872    let resolved_props = stride_delay_resolve_defaults(props);
873    let input = resolve(x);
874    let fb_signal = resolve(fb);
875
876    // Tap the feedback audio from the previous block.
877    let fb_audio = el::tap_in(serde_json::json!({"name": fbtap}));
878
879    // User processes the feedback audio.
880    let processed = insert(fb_audio);
881
882    // Apply feedback amount to processed return, sum with input.
883    let feedback_mix = el::mul((fb_signal, processed));
884    let summed_input = el::add((input, feedback_mix));
885
886    // Run delay with internal fb=0 (external loop handles feedback).
887    let delayed = Node::new(
888        "stridedelay",
889        resolved_props,
890        vec![resolve(delay_ms), resolve(0.0), summed_input],
891    );
892
893    // Tap the output for next block's feedback.
894    el::tap_out(serde_json::json!({"name": fbtap}), delayed)
895}

1008// ---- interpolateN (energy-preserving N-way crossfade) -----------------
1009
1010/// Energy-preserving N-way crossfading mixer.
1011///
1012/// Crossfades between a vector of N mono signal nodes using a normalised
1013/// interpolator index. All nodes are equally spaced along the
1014/// interpolator path.
1015///
1016/// ## Props
1017///
1018/// | Key         | Type | Default | Notes                                      |
1019/// |-------------|------|---------|--------------------------------------------|
1020/// | `barberpole`| bool | false   | Wrap the interpolator circularly (ring)    |
1021///
1022/// **Without barberpole** (default): interpolator is clamped to [0, 1].
1023/// `0` = first node, `1` = last node. Linear path.
1024///
1025/// **With barberpole**: nodes are on a circular ring. `0` and `1` both
1026/// map to the first node. The last node crossfades back into the first.
1027/// Values outside [0, 1] wrap seamlessly.
1028///
1029/// The crossfade uses the Signalsmith cheap energy-preserving curve:
1030/// `smoothstep(x) = 3x² − 2x³` passed through `sqrt` for equal-power
1031/// behaviour. See <https://signalsmith-audio.co.uk/writing/2021/cheap-energy-crossfade/>
1032///
1033/// # Arguments
1034///
1035/// - `props` — optional `{ "barberpole": true }`
1036/// - `interpolator` — position signal
1037/// - `nodes` — vector of mono signal nodes to crossfade between
1038///
1039/// Returns silence if fewer than 2 nodes are provided.
1040///
1041/// # Example
1042///
1043/// ```ignore
1044/// use elemaudio_rs::{el, extra};
1045/// use serde_json::json;
1046///
1047/// // Linear (clamped) crossfade between 3 oscillators
1048/// let mix = extra::interpolate_n(
1049///     json!({}),
1050///     el::const_with_key("xfade", 0.5),
1051///     vec![osc_a, osc_b, osc_c],
1052/// );
1053///
1054/// // Barberpole: wraps around, bipolar LFO is fine
1055/// let morph = extra::interpolate_n(
1056///     json!({ "barberpole": true }),
1057///     el::cycle(el::const_(0.1)),
1058///     vec![osc_a, osc_b, osc_c],
1059/// );
1060/// ```
1061pub fn interpolate_n(
1062    props: serde_json::Value,
1063    interpolator: impl Into<ElemNode>,
1064    nodes: Vec<impl Into<ElemNode>>,
1065) -> Node {
1066    let n = nodes.len();
1067    if n < 2 {
1068        log::error!("interpolate_n: requires at least 2 nodes, got {n}");
1069        return match nodes.into_iter().next() {
1070            Some(node) => resolve(node),
1071            None => el::const_(0.0),
1072        };
1073    }
1074
1075    let barberpole = props
1076        .get("barberpole")
1077        .and_then(|v| v.as_bool())
1078        .unwrap_or(false);
1079
1080    let resolved: Vec<Node> = nodes.into_iter().map(resolve).collect();
1081    let interp = resolve(interpolator);
1082
1083    if barberpole {
1084        interpolate_n_barberpole(interp, resolved, n)
1085    } else {
1086        interpolate_n_clamped(interp, resolved, n)
1087    }
1088}

1090/// Clamped linear interpolation: [0, 1] maps to [first, last].
1091fn interpolate_n_clamped(interp: Node, resolved: Vec<Node>, n: usize) -> Node {
1092    // Clamp to [0, 1]
1093    let clamped = el::min(1.0, el::max(0.0, interp));
1094
1095    // pos = clamped * (N - 1), maps [0,1] to [0, N-1]
1096    let pos = el::mul((clamped, (n - 1) as f64));
1097
1098    let weighted: Vec<Node> = resolved
1099        .into_iter()
1100        .enumerate()
1101        .map(|(i, node)| {
1102            let dist = el::abs(el::sub((pos.clone(), i as f64)));
1103            let proximity = el::max(0.0, el::sub((1.0, dist)));
1104            let ss = el::mul((
1105                el::mul((proximity.clone(), proximity.clone())),
1106                el::sub((3.0, el::mul((2.0, proximity)))),
1107            ));
1108            let gain = el::sqrt(ss);
1109            el::mul((node, gain))
1110        })
1111        .collect();
1112
1113    weighted
1114        .into_iter()
1115        .reduce(|acc, x| el::add((acc, x)))
1116        .unwrap_or_else(|| el::const_(0.0))
1117}

1119/// Barberpole wrapping: nodes on a circular ring, [0, 1) wraps.
1120fn interpolate_n_barberpole(interp: Node, resolved: Vec<Node>, n: usize) -> Node {
1121    // fract(x) = x - floor(x) — wraps any value to [0, 1)
1122    let fract = el::sub((interp.clone(), el::floor(interp)));
1123
1124    // pos = fract * N, maps [0, 1) to [0, N)
1125    let n_f = n as f64;
1126    let pos = el::mul((fract, n_f));
1127
1128    let weighted: Vec<Node> = resolved
1129        .into_iter()
1130        .enumerate()
1131        .map(|(i, node)| {
1132            let linear_dist = el::abs(el::sub((pos.clone(), i as f64)));
1133            // Circular distance: min(linear, N - linear)
1134            let circular_dist = el::min(linear_dist.clone(), el::sub((n_f, linear_dist)));
1135            let proximity = el::max(0.0, el::sub((1.0, circular_dist)));
1136            let ss = el::mul((
1137                el::mul((proximity.clone(), proximity.clone())),
1138                el::sub((3.0, el::mul((2.0, proximity)))),
1139            ));
1140            let gain = el::sqrt(ss);
1141            el::mul((node, gain))
1142        })
1143        .collect();
1144
1145    weighted
1146        .into_iter()
1147        .reduce(|acc, x| el::add((acc, x)))
1148        .unwrap_or_else(|| el::const_(0.0))
1149}

1151/// Sample-accurate one-shot 0→1 ramp.
1152///
1153/// On a rising edge of `x` (trigger crossing 0.5 upward), the signal increments
1154/// linearly from 0 to 1 over `dur` samples, then drops instantly back to 0 on
1155/// the next sample — hence the `00` suffix: the output starts at 0 and ends at
1156/// 0. Ideal as a sample-accurate envelope gate, a percussive modulator trigger,
1157/// or a duration-controlled one-shot LFO.
1158///
1159/// # Arguments (AGENTS.md order: props first, trigger/input last)
1160/// - `props` — see below
1161/// - `dur`  — ramp duration in **samples** (signal; may vary per-sample)
1162/// - `x`    — trigger signal; a rising edge through 0.5 starts the ramp
1163///
1164/// # Props
1165/// - `key`     : optional authoring key for stable identity
1166/// - `blocking`: `bool`, default `true`. When `true`, triggers are ignored
1167///   while the ramp is running (i.e. until the output returns to exactly 0).
1168///   When `false`, any rising edge restarts the ramp from 0.
1169///
1170/// # dur semantics
1171/// - `dur` is read every sample and the per-sample increment is `1 / dur`.
1172/// - If `dur` changes mid-ramp, the current value is preserved and only the
1173///   slope updates (smooth continuation at the new rate).
1174/// - If `dur <= 0` at the moment of a would-be trigger, the trigger is ignored.
1175/// - If `dur <= 0` while the ramp is running, the ramp aborts and the output
1176///   snaps to 0.
1177///
1178/// # Example
1179///
1180/// ```ignore
1181/// use elemaudio_rs::{el, extra};
1182/// use serde_json::json;
1183///
1184/// // 4800-sample (100 ms @ 48kHz) ramp, retriggered by a 2 Hz train,
1185/// // with retrigger blocking while the ramp is running.
1186/// let ramp = extra::ramp00(
1187///     json!({ "blocking": true }),
1188///     el::const_(4800.0),
1189///     el::train(el::const_(2.0)),
1190/// );
1191/// ```
1192pub fn ramp00(props: serde_json::Value, dur: impl Into<ElemNode>, x: impl Into<ElemNode>) -> Node {

1687/// Sparse random impulses with optional decaying release.
1688///
1689/// Inspired by SuperCollider's `Dust` with a twist: each impulse
1690/// can have a trailing exponential decay instead of being a single-sample
1691/// spike. Releases overlap and sum (polyphonic voice pool of 64).
1692///
1693/// # Arguments (AGENTS.md order: props first, inputs last)
1694/// - `props` — optional `seed`, `jitter`, and/or `key`
1695/// - `density` — impulses per second (Poisson rate, signal)
1696/// - `release` — T60 decay time in seconds per impulse (signal, audio-rate)
1697///
1698/// # Props
1699/// | Key       | Type | Default | Notes                                         |
1700/// |-----------|------|---------|-----------------------------------------------|
1701/// | `seed`    | num  | random  | Deterministic RNG seed (0 treated as 1)       |
1702/// | `jitter`  | num  | `0.0`   | Per-impulse amplitude randomness 0..1         |
1703/// | `key`     | str  | —       | Optional authoring key                        |
1704///
1705/// # Behaviour
1706/// - Each sample runs a Bernoulli trial with probability `density / sr`.
1707/// - On trigger, a new voice spawns with amplitude 1.
1708/// - `jitter` scales the impulse amplitude randomly per trigger.
1709/// - Voices decay exponentially at T60 = `release` seconds and sum at the output.
1710/// - If all 64 voice slots are busy, new triggers are dropped.
1711/// - `release <= 0` → single-sample impulse (voice expires immediately after firing).
1712/// - `density <= 0` → no new triggers, existing releases keep decaying.
1713///
1714/// # Overlap handling
1715/// New events use gap-filling spawn. When a trigger fires while the summed
1716/// envelope is at level `d`, the new voice is born at amplitude `(1 - d)` so
1717/// the envelope jumps back to exactly 1.0 rather than stacking on top.
1718/// A vendor-style `dcblock` poststage then recenters the output around 0.
1719/// Sonically this reads as a probabilistically-retriggered exponential
1720/// envelope whose decay tail shortens as density rises, without drifting DC.
1721///
1722/// For the `release <= 0` impulse mode, voices expire on the next sample so
1723/// there is no overlap to manage.
1724///
1725/// # Example
1726///
1727/// ```ignore
1728/// use elemaudio_rs::{el, extra};
1729/// use serde_json::json;
1730///
1731/// let noise = extra::rain(
1732///     json!({ "seed": 1, "jitter": 0.25 }),
1733///     el::const_(200.0),
1734///     el::const_(0.05),
1735/// );
1736/// ```
1737pub fn rain(
1738    props: serde_json::Value,
1739    density: impl Into<ElemNode>,
1740    release: impl Into<ElemNode>,
1741) -> Node {
1742    Node::new("rain", props, vec![resolve(density), resolve(release)])
1743}
